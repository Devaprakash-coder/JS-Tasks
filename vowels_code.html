      function vowels(words){
      const vowelarray = ["a", "e", "i", "o", "u"]; 
        var wordarray = words.split(""); //it will split the given string.
        var vowelcount=0;
          for (i=0; i<=wordarray.length;i++) //check the splited string one by one
          {
            if(vowelarray.includes(wordarray[i])) //if the vowels present add 1 to vowelcount
            {
            vowelcount++;
            }
          }
        return vowelcount;
    }
